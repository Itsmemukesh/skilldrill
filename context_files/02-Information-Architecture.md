    Home (Daily Training Dashboard)
    │
    ├── Continue Practice
    ├── Daily Challenge
    ├── Quick Practice

    ├── Random Challenge
    ├── Recent Performance
    └── Settings

    │   ├── Microsoft Writing Style Guide
    │   └── Google Developer Documentation Style Guide
    ├── Docs-as-Code
    ├── AI for Technical Writers
    ├── Content Strategy
    ├── Professional Skills
    └── Interview Preparation

    Quiz Flow
    │
    ├── Select Skill
    ├── Select 5 / 10 / 20 Questions
    ├── Select Difficulty
    ├── Start Quiz
    ├── Question Loop

    │   ├── Submit required
    │   ├── Show answer
    │   ├── Explanation
    │   ├── Reference
    │   ├── Real-world example
    │   └── Auto advance
    └── Results

    Settings
    │
    ├── Timer duration
    ├── Theme
    └── About
